    \section*{Probability}
    \begin{flalign*}
        P(A|B) & = \dfrac{P(A \cap B)}{P(B)} &  &
    \end{flalign*}
    If A and B are independent $P(A \cap B) = P(A)P(B)$
    \begin{flalign*}
        P(A|B) & = \dfrac{P(B|A)P(A)}{P(B)} &  &
    \end{flalign*}
    \subsection*{Expected Value}
    \begin{flalign*}
        E[x]    & = \int\limits_{-\infty}^{\infty} xp(x)\,dx    &  & \\
        E[g(x)] & = \int\limits_{-\infty}^{\infty} g(x)p(x)\,dx
    \end{flalign*}
    \subsection*{Variance}
    \begin{flalign*}
         & Var[x]        = E[(x-E[x])^2] = \sigma^2 = \int\limits_{-\infty}^{\infty} (x-E[x])^2 p(x)\,dx &  & \\
         & E((x-E[x])^2) = E[x^2] - (E[x])^2                                                             &  &
    \end{flalign*}
    \subsection*{Covariance}
    \begin{flalign*}
        cov(X_1, X_2) & = E[(X_1-m_1)(X_2-m_2)]  &  & \\
                      & = E[(X_1)(X_2)] - m_1m_2 &  &
    \end{flalign*}
    \subsection*{Correlation}
    \begin{flalign*}
        \rho & = \dfrac{cov(X_1, X_2)}{\sqrt{V(X_1)V(X_2)}} &  &
    \end{flalign*}
